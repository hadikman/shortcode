--- title: faker description: Faker function lets you generate dummy data for seeding the database. tags: laravel db date: 2019-08-12 ref: https://github.com/fzaninotto/Faker --- $faker = Faker\Factory::create(); $faker->name; // First and second name $faker->randomDigit; // A random number $faker->word; // A single word $faker->sentence; // A sentence $faker->unique()->word; // A single unique word $faker->text($maxNbChars = 300); // 300 character long text $faker->safeEmail; // An email address $faker->hexcolor; // Hex color
